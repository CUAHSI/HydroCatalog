CREATE VIEW [dbo].[LastServiceRecords]
AS
SELECT     a.*
FROM         dbo.ServiceMonitor AS a INNER JOIN
                          (SELECT     ServiceName, MAX(TimeChecked) AS maxid
                            FROM          dbo.ServiceMonitor
                            GROUP BY ServiceName) AS b ON a.TimeChecked = b.maxid

GO
